### Basic usage

```jsx
import { FormGroup, Label, Input } from 'beautiful-react-ui';

<FormGroup>
  <Label htmlFor="field-id">Field label</Label>
  <Input id="field-id" value="Input field" />
</FormGroup>
```

### Orientation

#### Horizontal

```jsx
import { FormGroup, Label, Input } from 'beautiful-react-ui';

<FormGroup orientation="horizontal">
  <Label htmlFor="field-id">Field label</Label>
  <Input id="field-id" value="Input field" />
</FormGroup>
```
#### Vertical

```jsx
import { FormGroup, Label, Input } from 'beautiful-react-ui';

<FormGroup orientation="vertical">
  <Label htmlFor="field-id">Field label</Label>
  <Input id="field-id" value="Input field" />
</FormGroup>
```
