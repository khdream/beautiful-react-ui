### Basic Usage

```jsx
import { DisplayField } from 'beautiful-react-ui';

<DisplayField label="Field label:" value="field value" />
```

### Label colors

```jsx
import { DisplayField } from 'beautiful-react-ui';

<>
  <DisplayField label="Primary label:" value="field value" labelColor="primary" />
  <DisplayField label="Secondary label:" value="field value" labelColor="secondary" />
  <DisplayField label="Info label:" value="field value" labelColor="info" />
  <DisplayField label="Success label:" value="field value" labelColor="success" />
  <DisplayField label="Warning label:" value="field value" labelColor="warning" />
  <DisplayField label="Danger label:" value="field value" labelColor="danger" />
</>
```

### Value colors

```jsx
import { DisplayField } from 'beautiful-react-ui';

<>
  <DisplayField label="Field label:" value="primary value" valueColor="primary" />
  <DisplayField label="Field label:" value="secondary value" valueColor="secondary" />
  <DisplayField label="Field label:" value="info value" valueColor="info" />
  <DisplayField label="Field label:" value="success value" valueColor="success" />
  <DisplayField label="Field label:" value="warning value" valueColor="warning" />
  <DisplayField label="Field label:" value="danger value" valueColor="danger" />
</>
```

### Border style

```jsx

<>
  <DisplayField label="Solid border:" value="field value" borderStyle="solid" />
  <DisplayField label="Dashes border:" value="field value" borderStyle="dashed" />
  <DisplayField label="Dotted border:" value="field value" borderStyle="dotted" />
  <DisplayField label="Double border:" value="field value" borderStyle="double" />
  <DisplayField label="No border:" value="field value" borderStyle="none" />
</>
```

### Bold style

