### Basic usage
```jsx
<Spinner />
```
### Spinner colors

```jsx
<Spinner />
<Spinner color="primary" />
<Spinner color="secondary" />
<Spinner color="info" />
<Spinner color="warning" />
<Spinner color="success" />
<Spinner color="danger" />
```

### Spinner types

```
<div>
  <Spinner type="pulse" />
  <Spinner type="pulse" color="primary" />
  <Spinner type="pulse" color="secondary" />
  <Spinner type="pulse" color="info" />
  <Spinner type="pulse" color="warning" />
  <Spinner type="pulse" color="success" />
  <Spinner type="pulse" color="danger" />
<div style={{marginTop: '20px'}}>
</div>
  <Spinner type="circle" />
  <Spinner type="circle" color="primary" />
  <Spinner type="circle" color="secondary" />
  <Spinner type="circle" color="info" />
  <Spinner type="circle" color="warning" />
  <Spinner type="circle" color="success" />
  <Spinner type="circle" color="danger" />
</div>
```
