### Basic usage

```jsx
import { Link } from 'beautiful-react-ui';

<Link href="#">Nunc lobortis diam nec elementum finibus.</Link>
```

### Color

```jsx
import { Link, Divider } from 'beautiful-react-ui';

<>
  <Link href="#" color="primary">A <strong>primary</strong> link</Link>
  <Divider line="dashed" />
  <Link href="#" color="secondary">A <strong>secondary</strong> link</Link>
  <Divider line="dashed" />
  <Link href="#" color="info">A <strong>info</strong> link</Link>
  <Divider line="dashed" />
  <Link href="#" color="success">A <strong>success</strong> link</Link>
  <Divider line="dashed" />
  <Link href="#" color="warning">A <strong>warning</strong> link</Link>
  <Divider line="dashed" />
  <Link href="#" color="danger">A <strong>danger</strong> link</Link>
</>
```
